def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    if city == 'NYC':
        time = datetime.strptime(datum['starttime'], '%m/%d/%Y %H:%M:%S')
    elif city == 'Chicago':
        time = datetime.strptime(datum['starttime'], '%m/%d/%Y %H:%M')      
    else:
        time = datetime.strptime(datum['Start date'], '%m/%d/%Y %H:%M')
        
    month = int(time.strftime('%m'))
    hour = int(time.strftime('%H'))
    day_of_week = time.strftime('%A')

    return (month, hour, day_of_week)

    assert time_of_trip(example_trips[city], city) == tests[city]

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)

        # 
        for row in trip_reader:
            # 
            new_point = {}
            
            ## ##
            
            ##  new_point  ##
            ##  DictWriter         ##
            new_point['duration'] = duration_in_mins(row, city)
            new_point['month'], new_point['hour'], new_point['day_of_week'] = time_of_trip(row, city)
            new_point['user_type'] = type_of_user(row, city)

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)

def number_of_trips(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
            else:
                n_customers += 1
        
        # 
        n_total = n_subscribers + n_customers
        # 
        rate_subscribers = '{:.1%}'.format(n_subscribers/n_total)
        # 
        rate_customers = '{:.1%}'.format(n_customers/n_total)
        # 
        return(n_total, rate_subscribers, rate_customers, )

def duration_time(filename):
    """
    
    30
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_overtime, overtime = 0, 0
        n_lesstime, lesstime = 0, 0
        
        # 
        for row in reader:
            if float(row['duration']) > 30:
                n_overtime += 1
                overtime += float(row['duration'])
            else:
                n_lesstime += 1
                lesstime += float(row['duration'])
        
        # 
        n_total = n_overtime + n_lesstime
        totaltime = overtime + lesstime
        #
        averagetime = totaltime/n_total
        # 30
        rate_overtime = '{:.1%}'.format(n_overtime/n_total)

        # 
        return(averagetime, rate_overtime)

def average_time(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers, totaltime_subscribers = 0, 0
        n_customers, totaltime_customers = 0, 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
                totaltime_subscribers += float(row['duration'])
            else:
                n_customers += 1
                totaltime_customers += float(row['duration'])
        
        # 
        average_subcribers = totaltime_subscribers/n_subscribers
        average_customers = totaltime_customers/n_customers

        # 
        return(average_subcribers, average_customers)
